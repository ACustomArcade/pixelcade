done

   echo -e "\n[FOUND] PixelcadeLCD"
   startup=true

   sudo fbi ~/pixelcade/system/lcd1024alert.png --noverbose -T 1 -d /dev/fb0 &
   ((start=`date +%s`))
   read -rsn 1 -t 5 -p "Waiting for user to quit or continueâ€¦";echo
   ((end=`date +%s`))
   ((total = end - start))
   if [ $total -gt 4 ]; then
      startup=true
      sudo killall fbi
   else
      startup=false
   fi
fi

   sudo killall fbi
